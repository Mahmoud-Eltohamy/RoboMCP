Getting Started

Installation

pip install mcp-appium

Basic Usage

from mcp_appium import client

# Initialize the client
appium_client = client.AppiumClient(
    appium_url="http://localhost:4723",
    capabilities={
        "platformName": "Android",
        "appium:automationName": "UiAutomator2",
        "appium:deviceName": "Android Emulator",
        "appium:app": "/path/to/your/app.apk"
    }
)

# Connect to the device
session = appium_client.create_session()

# Find and interact with elements
el = session.find_element("id", "login_button")
el.click()

# Take a screenshot
screenshot = session.get_screenshot()

# Close the session
session.quit()
