AI Capabilities

The AI can analyze the current screen of a mobile application to identify UI elements, understand layout structures, and detect interactive components.

from mcp_appium import client, ai_integration

# Initialize client and AI integration
appium_client = client.AppiumClient(...)
session = appium_client.create_session()
ai = ai_integration.MCPAIIntegration(provider="openai")

# Get screen description
screenshot = session.get_screenshot()
page_source = session.get_page_source()
description = ai.describe_screen(screenshot, page_source)

print(description)
# Output: "The screen shows a login form with username and password fields,
# a Login button, and a Forgot Password? link at the bottom..."

Generate complete test scripts based on natural language descriptions of testing goals. Support for multiple programming languages.

from mcp_appium import ai_integration

ai = ai_integration.MCPAIIntegration(provider="openai")

# Generate test script
test_goal = "Log in to the app, add an item to the cart, and verify it appears in the checkout screen"
language = "python"  # or "java", "javascript", "csharp", "ruby"

test_script = ai.generate_test_script(
    test_goal=test_goal,
    language=language,
    app_name="Shopping App"
)

print(test_script)
# Output: Complete Python test script with Appium code to perform the requested actions

Execute mobile testing actions using natural language commands without writing code manually.

from mcp_appium import client, ai_integration

# Initialize client and AI integration
appium_client = client.AppiumClient(...)
session = appium_client.create_session()
ai = ai_integration.MCPAIIntegration(provider="gemini")

# Execute commands with natural language
commands = [
    "Enter testuser in the username field",
    "Type password123 in the password field",
    "Click the login button",
    "Wait for the home screen to load",
    "Verify that the welcome message contains Hello, testuser"
]

for command in commands:
    actions = ai.interpret_command(command, session)
    execute_actions(session, actions)
